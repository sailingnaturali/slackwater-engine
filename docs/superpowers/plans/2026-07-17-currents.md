# Slackwater Currents Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Predict tidal currents (slack, max flood, max ebb) offline for XTide's US current stations, reusing the validated harmonic tide engine.

**Architecture:** A current station's harmonic math is identical to a tide station's — the same `evalH` sum-of-cosines, but the result is signed velocity (knots) along a flood axis instead of height. Max flood/ebb are slope-zeros (reuse `Extremes.swift`); **slack is the value-zero of velocity** — the one new finder. Subordinate stations warp a reference station's events by time/speed offsets (NOAA Current-Tables method). Station data is extracted from XTide's Harmbase2 SQL dump into a bundled `currents.json`.

**Tech Stack:** Swift 6 (SwiftPM, Swift Testing), Node build-time tools (`.mjs`), XTide (`tide -m c`) for validation goldens.

## Global Constraints

- **Runtime is zero-network.** XTide binary + SQL dump are used only at build/fixture-gen time, never at runtime. (Copied from spec.)
- **US NOAA current stations only.** No CHS/Canadian stations (deferred). (Spec: Scope.)
- **Current amplitudes are in knots**, carried opaquely by the constituent sum — the engine does no unit conversion; `evalH` returns knots directly for a current station.
- **Reuse over reinvention.** The prediction core (`ParamProvider`, `evalH`, `evalHPrime`, `findExtremes`) is validated; do not fork it. Widen visibility (`private` → internal) only where a new file must reuse it.
- **License header** on every new Swift file: `// TideEngine — MIT.`
- **Events-first for subordinate stations.** Continuous warped-curve reconstruction is deferred. (Spec: Non-goals.)
- Test framework is **Swift Testing** (`import Testing`, `@Test`, `#expect`, `#require`) — not XCTest.

```swift
// TideEngine — MIT. Tidal current prediction: velocity, max flood/ebb, and slack.
// Max flood/ebb are slope-zeros (reuses Extremes.findExtremes). Slack is the
// value-zero of velocity — the one genuinely new finder here.
import Foundation

private let slackToleranceHours = 1.0 / 3600  // 1 second

struct RawSlack { let hour: Double; let time: Date; let speed: Double }

/// Root of v(t) in [a, b] where v(a), v(b) have opposite signs. Bisection on evalH.
private func bisectZero(_ a0: Double, _ b0: Double, _ fa0: Double, _ params: [PreparedParam]) -> Double {
    var a = a0, b = b0, fa = fa0
    while true {
        let mid = (a + b) / 2
        if b - a < slackToleranceHours { return mid }
        let fMid = evalH(mid, params)
        if fMid == 0 { return mid }
        if fa > 0 ? fMid > 0 : fMid < 0 { a = mid; fa = fMid } else { b = mid }
    }
}

/// Slack instants (velocity value-zeros) in [fromHour, toHour]. Mirrors the
/// bracket-and-bisect structure of findExtremes, but roots evalH not evalHPrime.
func findSlacks(fromHour: Double, toHour: Double, provider: ParamProvider) -> [RawSlack] {
    var params = provider(max(0, fromHour))
    var lastGen = provider.generation
    if params.isEmpty { return [] }

    var maxSpeed = 0.0
    for p in params where p.w > maxSpeed { maxSpeed = p.w }
    if maxSpeed == 0 { return [] }
    let bracket = Double.pi / (2 * maxSpeed)

    var results: [RawSlack] = []
    var tPrev = fromHour
    var vPrev = evalH(tPrev, params)
    var tNext = tPrev + bracket
    while tNext <= toHour + bracket {
        let newParams = provider(tPrev)
        if provider.generation != lastGen { params = newParams; lastGen = provider.generation; vPrev = evalH(tPrev, params) }
        let tBound = min(tNext, toHour)
        let vNext = evalH(tBound, params)
        if vPrev != 0 && vNext != 0 && (vPrev > 0 ? vNext < 0 : vNext > 0) {
            let tRoot = bisectZero(tPrev, tBound, vPrev, params)
            if tRoot >= fromHour && tRoot <= toHour {
                results.append(RawSlack(hour: tRoot,
                                        time: Date(timeIntervalSince1970: provider.startMs / 1000 + tRoot * 3600),
                                        speed: evalH(tRoot, params)))
            }
        }
        if tBound >= toHour { break }
        tPrev = tBound
        vPrev = vNext
        tNext += bracket
    }
    return results
}
```

```swift
public struct CurrentPoint: Sendable { public let time: Date; public let speed: Double }
public enum CurrentEventKind: Sendable { case slack, maxFlood, maxEbb }
public struct CurrentEvent: Sendable {
    public let time: Date; public let speed: Double; public let kind: CurrentEventKind
}

/// A harmonic current station. Same constituent math as `Station`; the prediction
/// is signed velocity (knots) along the flood axis — positive = flood, negative = ebb.
public struct CurrentStation: Sendable {
    let constituents: [StationConstituent]
    let catalog: Catalog
    public let floodDirection: Double
    public let ebbDirection: Double

    public init(constituents inputs: [HarmonicConstituent],
                floodDirection: Double, ebbDirection: Double, offset: Double = 0) {
        let d2r = Double.pi / 180
        let cat = Catalog.shared
        self.catalog = cat
        self.floodDirection = floodDirection
        self.ebbDirection = ebbDirection
        var cs = inputs
            .filter { cat.entry($0.name) != nil }
            .map { StationConstituent(name: $0.name, amplitude: $0.amplitude, phase: d2r * $0.phase) }
        if offset != 0 { cs.append(StationConstituent(name: "Z0", amplitude: offset, phase: 0)) }
        self.constituents = cs
    }

    private func provider(from: Date, to: Date, step: Double = 600) -> (ParamProvider, Double) {
        let startSec = (from.timeIntervalSince1970 / step).rounded(.down) * step
        let endSec = (to.timeIntervalSince1970 / step).rounded(.up) * step
        let endHour = max(0, (endSec - startSec) / 3600)
        let base = astro(Date(timeIntervalSince1970: startSec))
        let p = ParamProvider(constituents: constituents, baseAstro: base, catalog: catalog,
                              startMs: startSec * 1000, endHour: endHour)
        return (p, endHour)
    }

    /// Signed velocity series (knots), sampled every `step` seconds.
    public func speeds(from: Date, to: Date, step: TimeInterval = 600) -> [CurrentPoint] {
        let timeline = makeTimeline(from: from, to: to, step: step)
        guard let first = timeline.items.first else { return [] }
        let base = astro(first)
        let p = ParamProvider(constituents: constituents, baseAstro: base, catalog: catalog,
                              startMs: timeline.startMs, endHour: timeline.endHour)
        return zip(timeline.items, timeline.hours).map { item, hour in
            CurrentPoint(time: item, speed: evalH(hour, p(hour)))
        }
    }

    /// Max flood (positive peak) and max ebb (negative peak) — slope-zeros of velocity.
    public func maxima(from: Date, to: Date) -> [CurrentEvent] {
        let (p, endHour) = provider(from: from, to: to)
        let raw = findExtremes(fromHour: 0, toHour: endHour, provider: p,
                               isDoubleTide: false, prominenceThreshold: 0.01)
        return raw.map { CurrentEvent(time: $0.time, speed: $0.level,
                                      kind: $0.high ? .maxFlood : .maxEbb) }
    }

    /// Slack water — velocity value-zeros.
    public func slacks(from: Date, to: Date) -> [CurrentEvent] {
        let (p, endHour) = provider(from: from, to: to)
        return findSlacks(fromHour: 0, toHour: endHour, provider: p)
            .map { CurrentEvent(time: $0.time, speed: $0.speed, kind: .slack) }
    }

    /// Slacks + maxima merged in time order — the app-facing event list.
    public func events(from: Date, to: Date) -> [CurrentEvent] {
        (slacks(from: from, to: to) + maxima(from: from, to: to))
            .sorted { $0.time < $1.time }
    }
}
```

- [ ] **Step 1: Write the golden generator**

Create `tools/gen-currents-golden.mjs`. It shells to XTide's `tide` for a current station, parses the plain-text events, and writes the fixture. XTide must be installed with free harmonics (see Step 2).

```javascript
// Generate a currents validation fixture from XTide's own `tide -m c` output.
// Usage: node tools/gen-currents-golden.mjs "<station name>" <start ISO> <end ISO> <out.json>
// Requires: xtide `tide` on PATH + free harmonics installed (HFILE_PATH set).
// The station's constituents come from `tide`'s harmonic dump (-m l is not it);
// we read them from the harmonics/SQL export separately and paste into the fixture —
// this tool fills only station meta + the event list; add `constituents` by hand
// (Task 5's extractor automates that later).
import { execFileSync } from 'node:child_process';
import { writeFileSync } from 'node:fs';

const [station, start, end, out] = process.argv.slice(2);
if (!out) { console.error('usage: gen-currents-golden.mjs "<station>" <startISO> <endISO> <out.json>'); process.exit(1); }

// `tide -m c` prints current events (slacks + max flood/ebb) as plain text lines.
const raw = execFileSync('tide', [
  '-l', station, '-b', toXtide(start), '-e', toXtide(end),
  '-m', 'c', '-f', 't', '-em', 'pSsMm', '-z'   // -z = UTC; -f t = text
], { encoding: 'utf8' });

const events = [];
for (const line of raw.split('\n')) {
  // Example line: "2026-06-01 03:12  2.9 knots, Flood  ..." or "... Slack, Ebb Begins"
  const m = line.match(/^(\d{4}-\d\d-\d\d)\s+(\d\d:\d\d)\s+(-?\d+\.\d+)?\s*knots?.*?(Flood|Ebb|Slack)/i);
  if (!m) continue;
  const [, date, hm, spd, what] = m;
  const w = what.toLowerCase();
  const kind = w === 'slack' ? 'slack' : (w === 'flood' ? 'maxFlood' : 'maxEbb');
  const speed = kind === 'slack' ? 0 : Number(spd) * (kind === 'maxEbb' ? -1 : 1);
  events.push({ time: `${date}T${hm}:00Z`, speed, kind });
}

writeFileSync(out, JSON.stringify({
  station, type: 'harmonic',
  floodDirection: null, ebbDirection: null, offset: 0,
  start, end, constituents: [], events
}, null, 2) + '\n');
console.error(`wrote ${out} — ${events.length} events. Fill constituents + directions by hand (or via Task 5).`);

function toXtide(iso) { return iso.replace('T', ' ').replace(':00Z', '').replace('Z', ''); }
```

```swift
import Foundation
import Testing
@testable import TideEngine

private struct CurrentGoldenFixture: Decodable {
    let station: String
    let floodDirection: Double
    let ebbDirection: Double
    let offset: Double
    let start: String
    let end: String
    let constituents: [C]
    let events: [E]
    struct C: Decodable { let name: String; let amplitude: Double; let phase: Double }
    struct E: Decodable { let time: String; let speed: Double; let kind: String }
}

/// Reproduce XTide's OWN `tide -m c` current events for a harmonic station.
/// Real-world tolerances like the tide gate: ±15 min on event time, ±0.2 kn on peaks.
@Test func harmonicCurrentMatchesXTide() throws {
    let fx = try loadFixture("currents-golden-harmonic", as: CurrentGoldenFixture.self)
    let station = CurrentStation(
        constituents: fx.constituents.map { HarmonicConstituent(name: $0.name, amplitude: $0.amplitude, phase: $0.phase) },
        floodDirection: fx.floodDirection, ebbDirection: fx.ebbDirection, offset: fx.offset
    )
    let computed = station.events(from: parseISO(fx.start).addingTimeInterval(-3600),
                                  to: parseISO(fx.end).addingTimeInterval(3600))
    #expect(!computed.isEmpty)

    var maxTimeErr = 0.0, maxSpeedErr = 0.0
    for e in fx.events {
        let t = parseISO(e.time)
        let kind: CurrentEventKind = e.kind == "slack" ? .slack : (e.kind == "maxFlood" ? .maxFlood : .maxEbb)
        let match = computed
            .filter { $0.kind == kind }
            .min { abs($0.time.timeIntervalSince1970 - t.timeIntervalSince1970) < abs($1.time.timeIntervalSince1970 - t.timeIntervalSince1970) }
        let m = try #require(match, "no computed \(e.kind) near \(e.time)")
        let timeErr = abs(m.time.timeIntervalSince1970 - t.timeIntervalSince1970) / 60
        maxTimeErr = max(maxTimeErr, timeErr)
        #expect(timeErr < 15, "\(e.kind) at \(e.time): time off \(timeErr) min")
        if kind != .slack {
            let speedErr = abs(abs(m.speed) - abs(e.speed))
            maxSpeedErr = max(maxSpeedErr, speedErr)
            #expect(speedErr < 0.2, "\(e.kind) at \(e.time): speed \(m.speed) vs \(e.speed), off \(speedErr) kn")
        }
    }
    print("harmonic current vs XTide — max time err \(maxTimeErr) min, max speed err \(maxSpeedErr) kn")
}
```

```swift
/// A subordinate current station: no constituents of its own. Its events are the
/// reference station's events, time-shifted per kind and speed-scaled for
/// flood/ebb (the NOAA Current-Tables reduction). Event list only — no curve.
public struct SubordinateStation: Sendable {
    let reference: CurrentStation
    public let slackTimeOffset: TimeInterval
    public let floodTimeOffset: TimeInterval
    public let ebbTimeOffset: TimeInterval
    public let floodSpeedRatio: Double
    public let ebbSpeedRatio: Double
    public let floodDirection: Double
    public let ebbDirection: Double

    public init(reference: CurrentStation,
                slackTimeOffset: TimeInterval, floodTimeOffset: TimeInterval, ebbTimeOffset: TimeInterval,
                floodSpeedRatio: Double, ebbSpeedRatio: Double,
                floodDirection: Double, ebbDirection: Double) {
        self.reference = reference
        self.slackTimeOffset = slackTimeOffset
        self.floodTimeOffset = floodTimeOffset
        self.ebbTimeOffset = ebbTimeOffset
        self.floodSpeedRatio = floodSpeedRatio
        self.ebbSpeedRatio = ebbSpeedRatio
        self.floodDirection = floodDirection
        self.ebbDirection = ebbDirection
    }

    public func events(from: Date, to: Date) -> [CurrentEvent] {
        // Widen the reference window so events that shift into [from,to] are included.
        let pad = max(abs(slackTimeOffset), max(abs(floodTimeOffset), abs(ebbTimeOffset))) + 3600
        let refEvents = reference.events(from: from.addingTimeInterval(-pad),
                                         to: to.addingTimeInterval(pad))
        let shifted = refEvents.map { e -> CurrentEvent in
            switch e.kind {
            case .slack:    return CurrentEvent(time: e.time.addingTimeInterval(slackTimeOffset), speed: 0, kind: .slack)
            case .maxFlood: return CurrentEvent(time: e.time.addingTimeInterval(floodTimeOffset), speed: e.speed * floodSpeedRatio, kind: .maxFlood)
            case .maxEbb:   return CurrentEvent(time: e.time.addingTimeInterval(ebbTimeOffset), speed: e.speed * ebbSpeedRatio, kind: .maxEbb)
            }
        }
        return shifted.filter { $0.time >= from && $0.time <= to }.sorted { $0.time < $1.time }
    }
}
```

```javascript
// Extract US current stations from XTide's Harmbase2 SQL dump → currents.json.
// Usage: node tools/gen-currents.mjs <harmbase2.sql> Sources/TideEngine/Resources/currents.json
// Harmbase2 tables (see flaterco.com harmbase2 schema): `data_sets` (station meta,
// units, flood/ebb dir, reference link, offsets) + `constituents` (per-station
// amplitude/phase). We take rows where units = 'knots' and country = 'United States'.
// This parses the dump's `COPY ... FROM stdin` blocks — no live Postgres needed.
import { readFileSync, writeFileSync } from 'node:fs';

const [sqlPath, out] = process.argv.slice(2);
if (!out) { console.error('usage: gen-currents.mjs <harmbase2.sql> <out.json>'); process.exit(1); }
const sql = readFileSync(sqlPath, 'utf8');

// Parse a COPY block into array-of-arrays keyed by the column list.
function copyBlock(table) {
  const re = new RegExp(`COPY ${table} \\(([^)]+)\\) FROM stdin;\\n([\\s\\S]*?)\\n\\\\\\.`, 'm');
  const m = sql.match(re);
  if (!m) throw new Error(`COPY block for ${table} not found`);
  const cols = m[1].split(',').map((c) => c.trim());
  const rows = m[2].split('\n').filter(Boolean).map((line) => {
    const vals = line.split('\t').map((v) => (v === '\\N' ? null : v));
    return Object.fromEntries(cols.map((c, i) => [c, vals[i]]));
  });
  return rows;
}

const dataSets = copyBlock('data_sets');
const consts = copyBlock('constituents'); // adjust table/column names to the actual dump

const usCurrents = dataSets.filter((d) => (d.units || '').includes('knot') && /United States/i.test(d.country || ''));

const byStation = new Map();
for (const c of consts) {
  if (!byStation.has(c.data_set_id)) byStation.set(c.data_set_id, []);
  if (Number(c.amplitude) > 0) byStation.get(c.data_set_id).push({ name: c.name, amplitude: Number(c.amplitude), phase: Number(c.phase) });
}

const slug = (s) => s.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/(^-|-$)/g, '').slice(0, 48);
const stations = usCurrents.map((d) => {
  const base = { id: slug(d.name), name: d.name, floodDirection: num(d.flood_dir), ebbDirection: num(d.ebb_dir) };
  if (d.reference_station_id) {
    return { ...base, type: 'subordinate', reference: slug(refName(dataSets, d.reference_station_id)),
      slackTimeOffset: sec(d.slack_offset), floodTimeOffset: sec(d.flood_offset), ebbTimeOffset: sec(d.ebb_offset),
      floodSpeedRatio: num(d.flood_ratio), ebbSpeedRatio: num(d.ebb_ratio) };
  }
  return { ...base, type: 'harmonic', offset: num(d.datum_offset) || 0, constituents: byStation.get(d.id) || [] };
});

writeFileSync(out, JSON.stringify({ note: 'Generated from XTide Harmbase2 SQL dump; US NOAA current stations only.', stations }, null, 0) + '\n');
console.error(`wrote ${out} — ${stations.length} US current stations`);

function num(v) { return v == null ? null : Number(v); }
function sec(v) { return v == null ? 0 : Math.round(Number(v) * 60); } // dump stores minutes
function refName(all, id) { return (all.find((x) => x.id === id) || {}).name || ''; }
```

> **Note for the implementer:** the exact Harmbase2 table/column names (`data_sets`, `constituents`, `flood_dir`, `reference_station_id`, offset units) must be confirmed against the actual dump — open it and match the `COPY` headers before trusting the field names above. The schema is documented at flaterco.com; adjust the field mapping, not the structure.

```swift
// TideEngine — MIT. Bundled US current-station catalog (from XTide Harmbase2).
import Foundation

public enum AnyCurrentStation: Sendable {
    case harmonic(CurrentStation)
    case subordinate(SubordinateStation)
    public func events(from: Date, to: Date) -> [CurrentEvent] {
        switch self {
        case .harmonic(let s): return s.events(from: from, to: to)
        case .subordinate(let s): return s.events(from: from, to: to)
        }
    }
}

public struct CurrentCatalog: Sendable {
    private let stations: [String: StationRecord]

    public static let shared = CurrentCatalog()

    private struct StationRecord: Decodable, Sendable {
        let id: String; let name: String; let type: String
        let floodDirection: Double?; let ebbDirection: Double?
        let offset: Double?
        let constituents: [Con]?
        let reference: String?
        let slackTimeOffset: Double?; let floodTimeOffset: Double?; let ebbTimeOffset: Double?
        let floodSpeedRatio: Double?; let ebbSpeedRatio: Double?
        struct Con: Decodable, Sendable { let name: String; let amplitude: Double; let phase: Double }
    }
    private struct File: Decodable { let stations: [StationRecord] }

    private init() {
        let url = Bundle.module.url(forResource: "currents", withExtension: "json")!
        // swiftlint:disable:next force_try
        let decoded = try! JSONDecoder().decode(File.self, from: Data(contentsOf: url))
        stations = Dictionary(uniqueKeysWithValues: decoded.stations.map { ($0.id, $0) })
    }

    public func ids() -> [String] { Array(stations.keys) }

    public func station(_ id: String) -> AnyCurrentStation? {
        guard let r = stations[id] else { return nil }
        if r.type == "harmonic" {
            return .harmonic(CurrentStation(
                constituents: (r.constituents ?? []).map { HarmonicConstituent(name: $0.name, amplitude: $0.amplitude, phase: $0.phase) },
                floodDirection: r.floodDirection ?? 0, ebbDirection: r.ebbDirection ?? 0, offset: r.offset ?? 0))
        }
        guard let refId = r.reference, let ref = stations[refId], ref.type == "harmonic" else { return nil }
        let reference = CurrentStation(
            constituents: (ref.constituents ?? []).map { HarmonicConstituent(name: $0.name, amplitude: $0.amplitude, phase: $0.phase) },
            floodDirection: ref.floodDirection ?? 0, ebbDirection: ref.ebbDirection ?? 0, offset: ref.offset ?? 0)
        return .subordinate(SubordinateStation(
            reference: reference,
            slackTimeOffset: r.slackTimeOffset ?? 0, floodTimeOffset: r.floodTimeOffset ?? 0, ebbTimeOffset: r.ebbTimeOffset ?? 0,
            floodSpeedRatio: r.floodSpeedRatio ?? 1, ebbSpeedRatio: r.ebbSpeedRatio ?? 1,
            floodDirection: r.floodDirection ?? 0, ebbDirection: r.ebbDirection ?? 0))
    }
}
```
